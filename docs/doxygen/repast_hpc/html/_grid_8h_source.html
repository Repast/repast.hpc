98  /**
99  * Moves the specifed object the specified distance from its current
100  * position along the specified angle. For example, <code>moveByVector(object, 1, Grid.NORTH)</code>
101  * will move the object 1 unit "north" up the y-axis, assuming a 2D grid. Similarly,
102  * <code>grid.moveByVector(object, 2, 0, Math.toRadians(90), 0)</code> will rotate 90
103  * degrees around the y-axis, thus moving the object 2 units along the z-axis.
104  * <p/>
105  * <b> Note that the radians / degrees are incremented in a anti-clockwise fashion, such that
106  * 0 degrees is "east", 90 degrees is "north", 180 is "west" and 270 is "south."
107  *
108  * @param agent the object to move
109  * @param distance the distance to move
110  * @param anglesInRadians the angle to move along in radians.
111  * @return a pair containing a bool that indicates whether the move was a success or not, and the
112  * point where the agent was moved to.
113  */
114  virtual std::pair<bool, Point<GPType> > moveByVector(const T* agent, double distance,
115  const std::vector<double>& anglesInRadians) = 0;

117  /**
118  * Moves the specified object from its current location by the specified
119  * amount. For example <code>moveByDisplacement(object, 3, -2, 1)</code>
120  * will move the object by 3 along the x-axis, -2 along the y and 1 along
121  * the z. The displacement argument can be less than the number of
122  * dimensions in the space in which case the remaining argument will be set
123  * to 0. For example, <code>moveByDisplacement(object, 3)</code> will move
124  * the object 3 along the x-axis and 0 along the y and z axes, assuming a 3D
125  * grid.
126  *
127  * @param agent
128  * the object to move
129  * @param displacement
130  * the amount to move the object
131  * @return a pair containing a bool that indicates whether the move was a success or not, and the
132  * point where the agent was moved to.
133  */
