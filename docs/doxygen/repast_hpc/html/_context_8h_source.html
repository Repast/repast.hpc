62 /**
63  * Unary function used in the transform_iterator that allows context iterators
64  * to return the agent maps values.

66 template<typename T>
67 struct SecondElement: public std::unary_function<typename boost::unordered_map<AgentId, boost::shared_ptr<T> >::value_type, boost::shared_ptr<T> > {
68  boost::shared_ptr<T> operator()(const typename boost::unordered_map<AgentId, boost::shared_ptr<T> >::value_type& value) const {
69  const boost::shared_ptr<T>& ptr = value.second;
70  return ptr;
71  }
72 };

110  /**
111  * Adds the agent to the context. Performs a check to ensure
112  * that no agent with the same ID (presumably the 'same' agent)
113  * has previously been added. If a matching ID is found, the
114  * new agent is not added, and the address of the pre-existing
115  * agent is returned. If no match is found, the agent is added
116  * and the return value is the same as the value passed
117  *
118  * @param agent the agent to add
119  *
120  * @return the address of the agent in the context; will be the
121  * same as the address passed if the agent was successfully added,
122  * but if there was already an agent with the same ID the address
123  * returned will be that of the pre-existing agent, which is
124  * not replaced.
125  */

745  /**
746  * Gets the projection information for all projections in this context, for all agents whose IDs are listed
747  * in the AgentRequest.
748  *
749  * The general sense of this method can be easily understood: given a list of agents, get the projection
750  * information for all of those agents. But there are some subtleties that should be kept in mind.
751  *
752  * "The projection information for an agent" is misleading. In fact, the projection information that
753  * is needed can vary depending on the context and on the kind of projection.
754  *
755  * Generally speaking, spaces return only one kind projection information: coordinate locations for
756  * the agent specified. This is the simplest case.
757  *
758  * The more complicated case is given by graphs. A graph projection can return different sets of information
759  * depending on how that information will have to be used. The basic issue is that a graph projection
760  * returns sets of edges, and edges must be connected to other agents; this means that a mechanism
761  * must be in place for ensuring that the projection info that arrives can be used, which means that
762  * for a given 'ego' agent, all 'alter' agents that are connected to it by edges must also be on the
763  * receiving process. (Note: Repast HPC 1.0 versions sent all of the alter agents' content along with
764  * the edge send; this version does not do this, partly to minimize the amount of information being
765  * packaged and sent but also because the alternative method used is integrated with the normal
766  * bookkeeping for sharing agent information across processes (AgentRequests).) In different
767  * circumstances, different assumptions can be made about what information will be available on the
768  * receiving process. Note that the coordinate information is generally referred to as 'Primary'
769  * information, while edge information is 'secondary'; in a third category ('secondary IDs') are the IDs of the
770  * alter agents, which can be packaged separately.
771  *
772  * The impact of this is that this function is generally called in the following ways:
773  *
774  * 1) When requesting agents: in this case, a copy of the agent will be sent from one
775  * process to another. No secondary information will be sent at all. This is because it is
776  * assumed that if an agent participated in a graph on the receiving process, it would already
777  * be present on that process and would not be being requested.
778  *
779  * 2) When synchronizing Projection Information: in this case, some secondary information
780  * (edges) is needed: the edges that connect the specified ego agent with edges on the receiving
781  * process. No secondary IDs are needed, because the only edges being sent are those that connect
782  * to agents on the receiving process, which will be assumed to already be available on that
783  * process.
784  *
785  * 3) When synchronizing Agent Status (moving agents from process to process): in this case,
786  * the full collection of projection information is needed, including all of the edges in which
787  * the specified agent participates and all of the secondary IDs. (The secondary
788  * IDs of agents that are already on the receiving process can be omitted, at least theoretically.)
789  * This allows the full reconstruction of Projection Information on the receiving process.
790  *
791  * @param req List of IDs for agents whose information is requested
792  * @param map A map into which the projection information will be placed. Key values represent the names
793  * of the projections in this context.
794  * @param secondaryInfo true if the 'secondary' projection info must also be returned
795  * @param secondaryIds A set of IDs for agents who are referred to by the projection informaton
796  * being returned (may be null)
797  * @param destProc The Process that will be receiving this information (the information sent may
798  * be customized depending on the destination process). If not specified a larger set of information
799  * will be sent.
800  */
801  void getProjectionInfo(AgentRequest req, std::map<std::string, std::vector<repast::ProjectionInfoPacket*> >& map,
802  bool secondaryInfo = false, std::set<AgentId>* secondaryIds = 0, int destProc = -1);
