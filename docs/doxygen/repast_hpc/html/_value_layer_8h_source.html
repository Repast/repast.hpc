294 ValueType& DiscreteValueLayer<ValueType, Borders>::get(const Point<int>& pt) {
295  std::vector<int> out(pt.dimensionCount());
296  borders.transform(pt.coords(), out);
297  if (_dense)

399 template<typename ValueType, typename Borders>
400 ValueType& ContinuousValueLayer<ValueType, Borders>::get(const Point<double>& pt) {
401  std::vector<double> out(pt.dimensionCount());
402  borders.transform(pt.coords(), out);
403  // need to insert do an insert and return reference to
404  // result so that assignment via [] uses a reference in
405  // the map. insert inserts the entry if it doesn't exist
406  // and returns the entry or the existing entry
407  std::pair<values_iter, bool> res = values.insert(std::make_pair(out, _defaultValue));
408  return res.first->second;
409 }
