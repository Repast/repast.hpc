355 template<typename T, typename CellAccessor, typename GPTransformer, typename Adder, typename GPType>
356 bool BaseGrid<T, CellAccessor, GPTransformer, Adder, GPType>::moveTo(const AgentId& id,
357  const std::vector<GPType>& newLocation) {
358  LocationMapIter iter = agentToLocation.find(id);
359 
360  if (iter == agentToLocation.end())
361  throw Repast_Error_2<AgentId>(id, Projection<T>::name()); // Agent has not yet been introduced to this space/is not present
362 
363  if (newLocation.size() < dimensions_.dimensionCount())
364  throw Repast_Error_3(newLocation.size(), dimensions_.dimensionCount()); // Destination not fully specified
365 
366  std::vector<GPType> transformedCoords(newLocation.size(), 0);
367  gpTransformer.transform(newLocation, transformedCoords);
368 
369  if (iter->second->point.coords() == transformedCoords) return true;
370  return doMove(transformedCoords, iter->second);
371 }

373 template<typename T, typename CellAccessor, typename GPTransformer, typename Adder, typename GPType>
374 bool BaseGrid<T, CellAccessor, GPTransformer, Adder, GPType>::doMove(std::vector<GPType>& location, GridPointHolder<T,
375  GPType>* gpHolder) {
376  Point<GPType> pt(location);
377  if (cellAccessor.put(gpHolder->ptr, pt)) {
378  if (gpHolder->inGrid) {
379  cellAccessor.remove(gpHolder->ptr, gpHolder->point);
380  } else {
381  size_++;
382  gpHolder->inGrid = true;
383  }
384  gpHolder->point = pt;
385  return true;
386  }
387  return false;
388 }

400 template<typename T, typename CellAccessor, typename GPTransformer, typename Adder, typename GPType>
401 std::pair<bool, Point<GPType> > BaseGrid<T, CellAccessor, GPTransformer, Adder, GPType>::moveByDisplacement(
402  const T* agent, const std::vector<GPType>& displacement) {
403  if (displacement.size() != dimensions_.dimensionCount())
404  throw Repast_Error_5(displacement.size(), dimensions_.dimensionCount()); // displacement vector must equal number of grid/space dimensions
405 
406  LocationMapIter iter = agentToLocation.find(agent->getId());
407  if (iter == agentToLocation.end())
408  throw Repast_Error_6<AgentId>(agent->getId(), Projection<T>::name()); // Agent has not in this grid / space
409 
410  GridPointHolder<T, GPType>* gpHolder = iter->second;
411  std::vector<GPType> newPos(displacement.size(), 0);
412  gpTransformer.translate(gpHolder->point.coords(), newPos, displacement);
413  std::vector<GPType> transformedCoords(newPos.size(), 0);
414  gpTransformer.transform(newPos, transformedCoords);
415  if (iter->second->point.coords() == transformedCoords)
416  return std::make_pair(true, Point<GPType> (transformedCoords));
417  return std::make_pair(moveTo(agent->getId(), transformedCoords), Point<GPType> (transformedCoords));
418 }

457 template<typename T, typename CellAccessor, typename GPTransformer, typename Adder, typename GPType>
458 double BaseGrid<T, CellAccessor, GPTransformer, Adder, GPType>::getDistanceSq(const Point<GPType>& pt1, const Point<
459  GPType>& pt2) const {
460  if (pt1.dimensionCount() != pt2.dimensionCount())
461  throw Repast_Error_7(pt1.dimensionCount(), pt2.dimensionCount()); // Points do not have same number of dimensions
462 
463  double sum = 0;
464  for (int i = 0, n = pt1.dimensionCount(); i < n; i++) {
465  double diff = pt1.getCoordinate(i) - pt2.getCoordinate(i);
466  if (gpTransformer.isPeriodic()) {
467  int dim = dimensions_.extents(i);
468  double absDiff = abs(diff);
469  if (absDiff > dim / 2) {
470  diff = dim - absDiff;
471  }
472  }
473  sum += diff * diff;
474  }
475 
476  return sum;
477 }

479 template<typename T, typename CellAccessor, typename GPTransformer, typename Adder, typename GPType>
480 void BaseGrid<T, CellAccessor, GPTransformer, Adder, GPType>::getDisplacement(const Point<GPType>& pt1, const Point<
481  GPType>& pt2, std::vector<GPType>& out) const {
482  if (pt1.dimensionCount() != pt2.dimensionCount())
483  throw Repast_Error_8(pt1.dimensionCount(), pt2.dimensionCount()); // Points do not have same number of dimensions
484 
485  for (int i = 0, n = pt1.dimensionCount(); i < n; i++) {
486  GPType diff = pt2.getCoordinate(i) - pt1.getCoordinate(i);
487  if (gpTransformer.isPeriodic()) {
488  int dim = dimensions_.extents(i);
489  GPType absDiff = abs(diff);
490  if (absDiff > dim / ((GPType) 2)) {
491  diff = dim - absDiff;
492  }
493  }
494  out[i] = diff;
495  }
496 }
