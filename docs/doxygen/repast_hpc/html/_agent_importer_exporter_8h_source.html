255 /**

257  * all previous requests for this agent will be

264 /**
265  * Checks an AgentId to see if it is marked as a 'full cancellation';
266  * if it is, MODIFIES IT back to its original, unmarked
267  * state and returns true
268  *
269  * @param id the AgentId to be checked; modified back to
270  * a normal AgentId if found to be a full cancellation

272  * as a full cancellation, false otherwise
273  */
274 bool checkForFullCancellation(AgentId& id){

276  id.currentRank((id.currentRank() + 1) * (-1));
277  return true;
278  }
279  else return false;

287 /**
288  * This class manages importing agent information; primarily
289  * this means constructing the appropriate mpi receives when
290  * agent information is to be exchanged. However, this class
291  * can also define specific semantics that can apply to
292  * agent requests- what to do in the case that an agent
293  * is requested twice, for example.
294  */
295 class AbstractImporter{
296 protected:
297  // Each child implementation is required to maintain this list
298  std::set<int> exportingProcesses;
299 
300 public:
301 
302  AbstractImporter();
303  virtual ~AbstractImporter();
304 
305  /**
306  * Gets a const reference to the set of ints representing the processes that are
307  * sending this process agent information
308  */
309  virtual const std::set<int>& getExportingProcesses(){ return exportingProcesses; }
310 
311  /**
312  * Given an agent request (including requests for agents on multiple other
313  * processes), makes a record of the agents that are being requested by this
314  * process and will therefore be received from other processes. The record must
315  * at a minimum indicate which other processes are sending agent information,
316  * but may include other information, such as how many times a particular
317  * agent has been requested.
318  */
319  virtual void registerOutgoingRequests(AgentRequest &req) = 0;
320 
321  /**
322  * Notifies this importer that the agent that it (presumably) has been importing
323  * has been removed from the simulation on its home process, and the information
324  * for that agent will no longer be sent
325  */
326  virtual void importedAgentIsRemoved(const AgentId& id) = 0;
327 
328  /**
329  * Notifies this importer that the agent that it (presumably) has been importing
330  * from another process has been moved; its information will now be received
331  * from its new home process (unless the agent was moved to this process)
332  */
333  virtual void importedAgentIsMoved(const AgentId& id, int newProcess) = 0;
334 
335  /**
336  * Some semantic sugar; operationally this is the same as 'importedAgentIsRemoved'
337  */
338  inline void importedAgentIsNowLocal(const AgentId& id){ importedAgentIsRemoved(id); }
339 
340  /**
341  * Get a printable indication of the data in this object
342  */
343  virtual std::string getReport() = 0;
344 
345  virtual void getSetOfAgentsBeingImported(std::set<AgentId>& set) = 0;
346 
347  virtual void clear(){
348  exportingProcesses.clear();
349  }
350 };

386 /**
387  * Importer that maintains a list of the agents being sent from
388  * each sending process. If there are no agents being sent the
389  * size of the list will be zero and no mpi 'receive' will be
390  * created; for non-zero-length lists a single mpi 'receiver'
391  * is created. An agent that is requested twice is
392  * placed in the list twice. Semantically this means that an
393  * agent cancellation will remove the agent from the list exactly
394  * once, but also that removing an agent from the list who is
395  * not found in the list will not reduce the list size (c.f. the
396  * 'count' version, where every cancellation reduces the count
397  * whether the specific agent being canceled was ever requested
398  * at all).
399  */
400 class Importer_LIST: public AbstractImporter{
401 
402 private:
403  std::map<int, std::list<AgentId>* > sources;
404 
405  // Inline helper functions to manage the internal recordkeeping
406 
407  // Gets the record associated with the specified other process;
408  // if none exists, one is created. 'exportingProcesses' is updated
409  inline std::list<AgentId>* getRecord(int rank);
410 
411  // Removes all the entries for a given agent from the specified record;
412  // Returns the number of entries removed
413  // Also: Invokes 'checkRecord'
414  inline int removeAll(const AgentId& id, std::list<AgentId>* record, int rank);
415 
416  // Removes one entry for a given ID (passes through to removeId by position)
417  inline void removeID(AgentId& id, std::list<AgentId>* record, int rank);
418 
419  // Removes the entry at the specified position in the record list.
420  // Also: Invokes 'checkRecord'
421  inline void removeID(std::list<AgentId>::iterator idPosition, std::list<AgentId>* record, int rank);
422 
423  // Checks the length of this record; if the record is now empty,
424  // it is removed from the map (and exportingProcesses is updated)
425  inline void checkRecord(std::list<AgentId>* record, int rank);
426 
427 public:
428  Importer_LIST();
429  ~Importer_LIST();
430 
431  virtual void registerOutgoingRequests(AgentRequest& req);
432  virtual void importedAgentIsRemoved(const AgentId& id);
433  virtual void importedAgentIsMoved(const AgentId& id, int newProcess);
434 
435  virtual std::string getReport();
436  virtual void getSetOfAgentsBeingImported(std::set<AgentId>& set);
437 
438  virtual void clear(){
439  AbstractImporter::clear();
440  sources.clear();
