212  /**
213  * Should return true if the Projection implemented can 'keep' some (non-local)
214  * agents during a projection information synchronization operation. Generally
215  * spaces will allow all non-local agents to be deleted, but graphs keep the
216  * non-local agents that participate in Master edges.
217  *
218  * It is possible to override these. A graph projection can be created that does
219  * not permit non-local agents to be 'kept'. This would be an extremely unusual
220  * use case, but it is possible.
221  *
222  * Note that these are used for optimization. If no projection in a given context
223  * keeps any agents, several steps in the synchronization algorithm can be omitted.
224  * Of course, omitting these steps when a projection actually retains agents can
225  * caused undefined problems.
226  *
227  * @return true if this projection will keep non-local agents during a projection
228  * information synchronziation event, false if it will not.
229  */
230  virtual bool keepsAgentsOnSyncProj() = 0;

232  /**
233  * Should return true if the Projection implemented will send secondary agents during
234  * a status exchange. Generally spaces do not and graphs do.
235  *
236  * If no secondary agents will be sent, portions of the algorithm can be omitted for
237  * optimization.
238  *

243  /**
244  * Gets the set of processes with which this Projection exchanges projection info.
245  * In the most general case this will be all other processors; this is the case
246  * for graphs, where agent connections can be arbitrary. However, spaces usually
247  * exchange information only with a small subset of 'neighbor' processes, which
248  * is knowable in advance and constant. To accommodate the general case, the
249  * algorithm for exchanging information must poll all other processes to see which
250  * are sending to this one; if this is known in advance, this additional (expensive)
251  * step can be skipped.

259  * be all other processors. However, simulations where agents move in spaces will

304 template<typename T>
305 void Projection<T>::getProjectionInfo(std::vector<AgentId>& agents, std::vector<ProjectionInfoPacket*>& packets,
306  bool secondaryInfo, std::set<AgentId>* secondaryIds, int destProc){
307  for(std::vector<AgentId>::const_iterator iter = agents.begin(), iterEnd = agents.end(); iter != iterEnd; iter++){
308  ProjectionInfoPacket* packet = getProjectionInfo((*iter), secondaryInfo, secondaryIds, destProc);
309  if((packet != 0) && (!packet->isEmpty())) packets.push_back(packet);
310  }
311 }
