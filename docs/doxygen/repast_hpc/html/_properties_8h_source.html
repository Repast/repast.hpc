240  /**
241  * Writes the contents of the properties file to the specified
242  * repast log (at 'INFO' log level)
243  *
244  * @param logName name of the log to use
245  * @param keysToWrite optional; if included, writes only the
246  * keys included in the vector and their values, in the
247  * order they appear in the vector. Will write
248  * blank values for any key name in the vector that is not
249  * in the properties file. If not included, all properties
250  * and their values are written, in map order.
251  */
252  void log(std::string logName, std::vector<std::string> *keysToWrite = 0){
253  if(keysToWrite != 0){
254  std::vector<std::string>::iterator iter = keysToWrite->begin();
255  std::vector<std::string>::iterator iterEnd = keysToWrite->end();
256  while(iter != iterEnd){
257  Log4CL::instance()->get_logger(logName).log(INFO, (*iter) + " = " + (getProperty(*iter)));
258  iter++;
259  }
260  }
261  else{
262  std::map<std::string, std::string>::iterator iter = map.begin();
263  std::map<std::string, std::string>::iterator iterEnd = map.end();
264  while(iter != iterEnd){
265  Log4CL::instance()->get_logger(logName).log(INFO, iter->first + " = " + iter->second);
266  iter++;
267  }
268  }
269  }
