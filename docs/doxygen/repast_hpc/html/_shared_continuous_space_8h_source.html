120 template<typename T, typename GPTransformer, typename Adder>
121 template<typename AgentContent, typename ContentProvider>
122 void SharedContinuousSpace<T, GPTransformer, Adder>::sendToNeighbor(GridBufferSyncher<AgentContent, double>& syncher,
123  Neighbors::Location location, ContentProvider& provider, int val) {
124 
125  Neighbor* ngh = SharedBaseGridType::nghs.neighbor(location);
126  if (ngh != 0) {
127  std::vector<CellContents<AgentContent, double> > toSend;
128  GridDimensions bounds = SharedBaseGridType::createSendBufferBounds(location);
129  // check each agent if its within the bounds and if so get its content etc.
130  // TODO this is crazy inefficient -- need a KDTree or something similar
131  // container, so can get range out of that together with their points
132  for (typename SharedBaseGridType::GridBaseType::LocationMapConstIter iter =
133  SharedBaseGridType::GridBaseType::locationsBegin(); iter
134  != SharedBaseGridType::GridBaseType::locationsEnd(); ++iter) {
135  GridPointHolder<T, double> *gp = iter->second;
136  if (gp->inGrid && bounds.contains(gp->point)) {
137  CellContents<AgentContent, double> contents(gp->point);
138  provider.provideContent(gp->ptr.get(), contents._objs);
139  toSend.push_back(contents);
140  }
141  }
142 
143  syncher.send(ngh->rank(), toSend, val);
144  }
145 }

147 template<typename T, typename GPTransformer, typename Adder>
148 template<typename AgentContent, typename ContentProvider, typename ContentReceiver>
149 void SharedContinuousSpace<T, GPTransformer, Adder>::synchBuffer(SharedContext<T>& context, ContentProvider& provider,
150  ContentReceiver& receiver) {
151 
152  SharedBaseGridType::buffered.clear();
153 
154  GridBufferSyncher<AgentContent, double> syncher(SharedBaseGridType::comm);
155 
156  syncher.receive(SharedBaseGridType::nghs.neighbor(Neighbors::E), GRID_BUFFER_SYNC0);
157  sendToNeighbor<AgentContent> (syncher, Neighbors::W, provider, GRID_BUFFER_SYNC0);
158 
159  syncher.receive(SharedBaseGridType::nghs.neighbor(Neighbors::W), GRID_BUFFER_SYNC1);
160  sendToNeighbor<AgentContent> (syncher, Neighbors::E, provider, GRID_BUFFER_SYNC1);
161 
162  syncher.receive(SharedBaseGridType::nghs.neighbor(Neighbors::N), GRID_BUFFER_SYNC2);
163  sendToNeighbor<AgentContent> (syncher, Neighbors::S, provider, GRID_BUFFER_SYNC2);
164 
165  syncher.receive(SharedBaseGridType::nghs.neighbor(Neighbors::S), GRID_BUFFER_SYNC3);
166  sendToNeighbor<AgentContent> (syncher, Neighbors::N, provider, GRID_BUFFER_SYNC3);
167 
168  syncher.receive(SharedBaseGridType::nghs.neighbor(Neighbors::NE), GRID_BUFFER_SYNC4);
169  sendToNeighbor<AgentContent> (syncher, Neighbors::SW, provider, GRID_BUFFER_SYNC4);
170 
171  syncher.receive(SharedBaseGridType::nghs.neighbor(Neighbors::NW), GRID_BUFFER_SYNC5);
172  sendToNeighbor<AgentContent> (syncher, Neighbors::SE, provider, GRID_BUFFER_SYNC5);
173 
174  syncher.receive(SharedBaseGridType::nghs.neighbor(Neighbors::SE), GRID_BUFFER_SYNC6);
175  sendToNeighbor<AgentContent> (syncher, Neighbors::NW, provider, GRID_BUFFER_SYNC6);
176 
177  syncher.receive(SharedBaseGridType::nghs.neighbor(Neighbors::SW), GRID_BUFFER_SYNC7);
178  sendToNeighbor<AgentContent> (syncher, Neighbors::NE, provider, GRID_BUFFER_SYNC7);
179 
180  syncher.wait();
181 
182  for (size_t i = 0; i < syncher.vecsSize(); ++i) {
183  std::vector<CellContents<AgentContent, double> >* contentsList = syncher.received(i);
184  int rank = syncher.nghRank(i);
185  for (size_t j = 0; j < contentsList->size(); ++j) {
186  CellContents<AgentContent, double> contents = (*contentsList)[j];
187  std::vector<T*> out;
188  receiver.createAgents(contents._objs, out);
189  for (typename std::vector<T*>::iterator iter = out.begin(); iter != out.end(); ++iter) {
190  T* obj = *iter;
191  obj->getId().currentRank(rank);
192  AgentId id = obj->getId();
193  SharedBaseGridType::buffered.push_back(id);
194  if (context.addAgent(obj) != obj) {
195  // already exists in the context so delete this one.
196  // this will occur if an obj is in the buffer of > 1
197  // spaces in the same context
198  delete obj;
199  }
200  context.incrementProjRefCount(id);
201  synchMoveTo(id, contents._pt);
202  }
203  }
204  }
205 }
