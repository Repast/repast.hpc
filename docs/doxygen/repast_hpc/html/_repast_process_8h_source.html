87 /**
88  * Contains information sent as agents are exchanged, either in response to
89  * requests or agent movement. Contains both agent raw information
90  * (of type 'Content') and projection information.
91  *
92  * Note: A 'Packet' is responsible for deleting the objects to which it points
93  * This is essentially not optional: when boost sends the Packet via MPI the locations
94  * at which it places the different elements are not known (no 'new' is called
95  * in the user code). Some code must be written to track these down and delete,
96  * and it is manifestly easier to provide that code in the Packet itself
97  * than to rewrite where needed, inspecting the Packet for the locations
98  */
99 template<typename Content>
100 class Request_Packet {
101 
102  friend class boost::serialization::access;
103 
104 public:
105  std::vector<Content>* agentContentPtr;
106  std::map<std::string, std::vector<ProjectionInfoPacket*> >* projectionInfoPtr;
107 
108  Request_Packet(): agentContentPtr(0), projectionInfoPtr(0){ }
109 
110  Request_Packet(std::vector<Content>* agentContent, std::map<std::string, std::vector<ProjectionInfoPacket*> >* projectionInfo):
111  agentContentPtr(agentContent), projectionInfoPtr(projectionInfo){
112  }
113 
114  ~Request_Packet(){
115  delete agentContentPtr;
116  agentContentPtr = 0;
117 
118  if(projectionInfoPtr != 0){
119  for(std::map<std::string, std::vector<ProjectionInfoPacket*> >::iterator iter = projectionInfoPtr->begin(), iterEnd = projectionInfoPtr->end(); iter != iterEnd; ++iter){
120  for(std::vector<ProjectionInfoPacket*>::iterator PIPIter = iter->second.begin(), PIPIterEnd = iter->second.end(); PIPIter != PIPIterEnd; ++PIPIter){
121  delete *PIPIter;
122  }
123  }
124  }
125  delete projectionInfoPtr;
126  projectionInfoPtr = 0;
127 
128  }
129 
130  template<class Archive>
131  void serialize(Archive& ar, const unsigned int version) {
132  ar & agentContentPtr;
133  ar & projectionInfoPtr;
134  }

410  // Repast Process should handle four specific tasks for controlling parallelization:
411  //
412  // 1. Requesting agents that are to be shared
413  // 2. Synchronizing the state values for shared agents
414  // 3. Synchronizing the Projection info for all shared agents
415  // 4. Synchronizing agent status values, including removing agents and moving them across processes
416  //
417  // At the end of all four of these operations, the simulation should be in a valid state
418  // (Though not necessarily a 'current' state: synchronizing state values does NOT update Projection
419  // information. This is a performance consideration: ideally we would always have the simulation in
420  // a 100% current state, but because the simulation is being parallelized, a choice must always be made
421  // about what information is updated.)

480 /**
481  * Requests agents from one process to others. Copies of the requested agents'
482  * Content are retrieved from their respective processes, created using
483  * the AgentCreator and added to the specified context.
484  *
485  * @param context the context to which the requested agents will be added
486  * @param request the AgentRequest containing the ids of the requested agents
487  * @param provider provides Content for a given an AgentRequest
488  * @param creator creates agents of type T given Content.
489  *
490  *
491  * @tparam T the type of the agents in the context
492  * @tparam Content the serializable struct or class that describes the
493  * state of agents
494  * @tparam Provider given an AgentRequest, a Provider provides the Content
495  * for the requested agents, implementing void provideContent(const AgentRequest&,
496  * std::vector<Content>&)
497  * @tparam AgentCreator a class that can create agents from Content, implementing
498  * T* createAgent(Content&).
499  *
500  */
501 template<typename T, typename Content, typename Provider, typename Updater, typename AgentCreator>
502 void RepastProcess::requestAgents(SharedContext<T>& context, AgentRequest& request, Provider& provider, Updater& updater, AgentCreator& creator
503 #ifdef SHARE_AGENTS_BY_SET
504  , std::string setName, AGENT_IMPORTER_EXPORTER_TYPE setType
505 #endif
506 ){
507 
508 
509  // Initiate the new requests
510 #ifdef SHARE_AGENTS_BY_SET
511  initiateAgentRequest(request, setName, setType);
512 #else
513  initiateAgentRequest(request);
514 #endif
515 
516  // Establish which processes are sending to/receiving from this one
517 #ifdef SHARE_AGENTS_BY_SET
518  const std::set<int>& exporters = importer_exporter->getExportingProcesses(setName);
519  const std::map<int, AgentRequest>& agentsToExport = importer_exporter->getAgentsToExport(setName);
520 #else
521  const std::set<int>& exporters = importer_exporter->getExportingProcesses();
522  const std::map<int, AgentRequest>& agentsToExport = importer_exporter->getAgentsToExport();
523 #endif
524 
525 
526  // Construct MPI requests (Receives and Sends)
527  std::vector<boost::mpi::request> requests; // MPI Requests (receives and sends)
528 
529 
530  // Construct Receives
531  std::vector<Request_Packet<Content>* > toReceive;
532 
533  for (std::set<int>::const_iterator iter = exporters.begin(); iter != exporters.end(); ++iter) {
534  Request_Packet<Content>* packet;
535  toReceive.push_back( packet = new Request_Packet<Content>() );
536  requests.push_back(world->irecv(*iter, 23, *packet));
537  }
538 
539 
540  // Construct Sends
541  boost::ptr_list<Request_Packet<Content> >* toSend = new boost::ptr_list<Request_Packet<Content> >;
542 
543  for (std::map<int, AgentRequest>::const_iterator iter = agentsToExport.begin(); iter != agentsToExport.end(); ++iter) {
544 
545  // Agent Content
546  std::vector<Content>* content = new std::vector<Content>;
547  provider.provideContent(iter->second, *content);
548 
549  // Projection Info
550  std::map<std::string, std::vector<ProjectionInfoPacket*> >* projInfo = new std::map<std::string, std::vector<repast::ProjectionInfoPacket*> >;
551  context.getProjectionInfo(iter->second, *projInfo);
552 
553  Request_Packet<Content>* packet;
554  toSend->push_back( packet = new Request_Packet<Content>(content, projInfo) );
555  requests.push_back(world->isend(iter->first, 23, *packet));
556  }
557 
558  // Wait until all sends/receives complete
559  boost::mpi::wait_all(requests.begin(), requests.end());
560 
561  // Clear sent data
562  delete toSend;
563 
564  // Process (and delete) received data
565  for (typename std::vector<Request_Packet<Content>* >::iterator iter = toReceive.begin(), iterEnd = toReceive.end(); iter != iterEnd; ++iter) {
566  std::vector<Content>* content = (*iter)->agentContentPtr;
567  for (typename std::vector<Content>::const_iterator contentIter = content->begin(),
568  contentIterEnd = content->end(); contentIter != contentIterEnd; ++contentIter) {
569  T* out = creator.createAgent(*contentIter);
570  T* inContext = context.addAgent(out);
571  if (inContext != out){ // This agent was already on this process
572  updater.updateAgent(*contentIter);
573  delete out;
574  }
575  }
576  context.setProjectionInfo(*((*iter)->projectionInfoPtr));
577  delete *iter;
578  }
579

584 template<typename Content, typename Provider, typename Updater>
585 void RepastProcess::synchronizeAgentStates(Provider& provider, Updater& updater
586 #ifdef SHARE_AGENTS_BY_SET
587  , std::string setName
588 #endif
589 ){
590 
591  // Establish which processes are sending/receiving from this one
592 #ifdef SHARE_AGENTS_BY_SET
593  const std::set<int>& processesToReceiveFrom = importer_exporter->getExportingProcesses(setName);
594  const std::map<int, AgentRequest>& agentsToExport = importer_exporter->getAgentsToExport(setName);
595 #else
596  const std::set<int>& processesToReceiveFrom = importer_exporter->getExportingProcesses();
597  const std::map<int, AgentRequest>& agentsToExport = importer_exporter->getAgentsToExport();
598 #endif
599 
600 
601  // Construct MPI Requests (sends and receives)
602  std::vector<boost::mpi::request> requests;
603 
604  // Construct Receives
605  std::vector<std::vector<Content>*> received;
606  for (std::set<int>::const_iterator iter = processesToReceiveFrom.begin(), iterEnd = processesToReceiveFrom.end(); iter != iterEnd; ++iter) {
607  std::vector<Content>* content = new std::vector<Content>();
608  requests.push_back(world->irecv(*iter, 47, *content));
609  received.push_back(content);
610  }
611 
612  // Construct Sends
613  boost::ptr_list<std::vector<Content> >* toSend = new boost::ptr_list<std::vector<Content> >;
614  std::vector<Content>* content;
615 
616  for (std::map<int, AgentRequest>::const_iterator iter = agentsToExport.begin(), iterEnd = agentsToExport.end(); iter != iterEnd; ++iter) {
617  toSend->push_back( content = new std::vector<Content> );
618  provider.provideContent(iter->second, *content);
619  requests.push_back(world->isend(iter->first, 47, *content));
620  }
621 
622  // Wait until all sends and receives are complete
623  boost::mpi::wait_all(requests.begin(), requests.end());
624 
625  // Clear sent data
626  delete toSend;
627 
628  // Process (and clear) received data
629  for(typename std::vector<std::vector<Content>*>::iterator iter = received.begin(), iterEnd = received.end(); iter != iterEnd; ++iter){
630  content = *iter;
631  for (typename std::vector<Content>::const_iterator agentIter = content->begin(), agentIterEnd = content->end();
632  agentIter != agentIterEnd; ++agentIter){
633  updater.updateAgent(*agentIter);
634  }
635  delete content;
636  }
637 
638 }

642 template<typename T, typename Content, typename Provider, typename Updater, typename AgentCreator>
643 void RepastProcess::synchronizeProjectionInfo(SharedContext<T>& context, Provider& provider, Updater& updater, AgentCreator& creator,
644  EXCHANGE_PATTERN exchangePattern
645 #ifdef SHARE_AGENTS_BY_SET
646  , bool declareNoAgentsKeptOnAnyProcess
647 #endif
648  ){
649 
650  // Generate sets of agents to delete or not delete
651  std::set<AgentId> agentsToKeep;
652 
653  bool agentsMayBeKept =
654 #ifdef SHARE_AGENTS_BY_SET
655  context.keepsAgentsOnSyncProj() || !declareNoAgentsKeptOnAnyProcess;
656 #else
657  context.keepsAgentsOnSyncProj();
658 #endif
659 
660 
661  // If 'By Set': Construct 'Keep list' from all non-default I/E requests (agents being imported)
662 #ifdef SHARE_AGENTS_BY_SET
663  if(declareNoAgentsKeptOnAnyProcess){
664  importer_exporter->clear();

832 void RepastProcess::synchronizeAgentStatus(SharedContext<T>& context, Provider& provider, Updater& updater, AgentCreator& creator,
833  EXCHANGE_PATTERN exchangePattern) {
834 
835  // Step 1: Exchange information about agents whose status will be updated.
836  //
837  // Status Updates have been added to the importer_exporter and can now be exchanged
838  // across processes. All processes will now know that an agent that has previously
839  // been managed by process A will either be eliminated from the simulation or
840  // moved to process B.
841  std::vector<std::vector<AgentStatus>* > statusUpdates;
842  importer_exporter->exchangeAgentStatusUpdates(*world, statusUpdates);
843 
844  for (size_t i = 0, n = statusUpdates.size(); i < n; i++) {
845  std::vector<AgentStatus>* vec = statusUpdates[i];
846  for (size_t j = 0, k = vec->size(); j < k; ++j) {
847  AgentStatus& status = (*vec)[j];
848  if (status.getStatus() == AgentStatus::REMOVED) {
849  importer_exporter->importedAgentIsRemoved(status.getOldId()); // Notify importer/exporter that this agent will not be imported anymore
850  context.importedAgentRemoved(status.getOldId()); // Remove from context; agent cannot exist on this process after removal from home process
851  }
852  else if (status.getStatus() == AgentStatus::MOVED) {
853  if (rank_ != status.getNewId().currentRank()) {
854  // Notify importer that this agent will not be imported from the original
855  // process, but will instead be imported from its new home
856  importer_exporter->importedAgentIsMoved(status.getOldId(), status.getNewId().currentRank());
857  // Find it and update its id
858  T* agent = context.getAgent(status.getOldId());
859  if (agent == (void*) 0) throw Repast_Error_32<AgentId>(status.getOldId()); // Agent not found
860  agent->getId().currentRank(status.getNewId().currentRank());
861  }
862  else{
863  // Notify importer that the agent will not be imported anymore because this is its home process now
864  importer_exporter->importedAgentIsNowLocal(status.getOldId());
865  }
866  }
867  }
868  delete vec;
869  }
870 
871 
872  // Step 2: Send moving agents' information to new home processes
873  //
874  // First, some basic data structures must be created for some bookkeeping we will need later
875  std::set<AgentId> agentsToDrop; // A list of agents that will be removed from this process
876  std::set<int> psMovedTo; // A list of the processes that will be receiving moving agents
877  std::map<int, AgentRequest> agentRequests; // A map of these receiving processes and a list of the IDs of the agents going to them
878 
879 
880  // This loop applies to all agents moving off of this process (from 'movedAgents'); it:
881  // Re-sets the 'current Process' ID values of the agents moving away from this process to their new values
882  // Adds the agents to the lists of agents to be removed from this process
883  // Creates the map of agent requests per receiving process and adds the agents to it
884  for (MovedAgentSetType::const_iterator iter = movedAgents.begin(), iterEnd = movedAgents.end(); iter != iterEnd; ++iter) {
885  AgentId id = *iter;
886  context.getAgent(id)->getId().currentRank(id.currentRank());
887  agentsToDrop.insert(id);
888  int currentProc = id.currentRank();
889  if(psMovedTo.insert(currentProc).second){
890  AgentRequest req = AgentRequest(currentProc, rank_);
891  req.addRequest(id);
892  agentRequests[currentProc] = req;
893  }
894  else agentRequests[currentProc].addRequest(id);
895  }
896  movedAgents.clear();
897 
898  // Next, coordinate the send/receive pairs (which processes send to which)
899  //
900  // Three different methods can be used:
901  // 1) POLL: Assume no knowledge of which processes will be sending to this one; do a full exchange
902  // where all processes inform all other processes of whether they need to send to them
903  // 2) USE_CURRENT: Assume that the context (and, by implication, all projections in the context)
904  // know which processes they need to send to and which they will receive from. For example, a grid
905  // projection will know its 8 neighbors; a graph projection _might_ know which processes own
906  // nodes to which it is connected
907  // 3) USE_LAST: Assume that the set of processes to which sends go and from which sends are received
908  // will be unchanged from the last time this loop was run; once established, just keep using
909  // the same set
910  //
911  // Variants 'USE_LAST_OR_POLL' and 'USE_LAST_OR_USE_CURRENT' allow for the case in which one method (either
912  // POLL or USE_CURRENT) is used for the initial pass through the loop and thereafter the sets are assumed
913  // unchanged.
914  std::vector<int> psToSendTo; // Convert set to vector
915  std::vector<int> psToReceiveFrom;
916 
917  if(exchangePattern == USE_CURRENT || ((exchangePattern == USE_LAST_OR_USE_CURRENT) && (procsToSendAgentStatusInfoTo == NULL))){
918  std::set<int> sends, recvs;
919  context.getAgentStatusInfoExchangePartners(sends, recvs);
920  psToSendTo.assign(sends.begin(), sends.end());
921  psToReceiveFrom.assign(recvs.begin(), recvs.end());
922  // Add dummy requests to create empty sends
923  for(std::vector<int>::iterator iter = psToSendTo.begin(), iterEnd = psToSendTo.end(); iter != iterEnd; ++iter){
924  int dest = *iter;
925  if(agentRequests.find(dest) == agentRequests.end()){
926  AgentRequest dummy(rank_, dest);
927  agentRequests[dest] = dummy;
928  }
929  }
930  }
931  else if(exchangePattern == POLL || ((exchangePattern == USE_LAST_OR_POLL) && (procsToSendAgentStatusInfoTo == NULL))){
932  for(std::map<int, AgentRequest>::const_iterator iter = agentRequests.begin(), iterEnd = agentRequests.end(); iter != iterEnd; ++iter){
933  psToSendTo.push_back(iter->first);
934  }
935  SRManager manager(world);
936  manager.retrieveSources(psToSendTo, psToReceiveFrom, AGENT_MOVED_SENDERS);
937  }
938  else{
939  psToSendTo.assign(procsToSendAgentStatusInfoTo->begin(), procsToSendAgentStatusInfoTo->end());
940  psToReceiveFrom.assign(procsToRecvAgentStatusInfoFrom->begin(), procsToRecvAgentStatusInfoFrom->end());

942  // Add dummy requests to create empty sends
943  for(std::vector<int>::iterator iter = psToSendTo.begin(), iterEnd = psToSendTo.end(); iter != iterEnd; ++iter){
944  int dest = *iter;
945  if(agentRequests.find(dest) == agentRequests.end()){
946  AgentRequest dummy(rank_, dest);
947  agentRequests[dest] = dummy;
948  }
949  }
950  }
951 
952  saveAgentStatusInfoSRProcs(psToSendTo, psToReceiveFrom);
953 
954 
955  // Determine if any projection in the context will need to send 'secondary' agent data:
956  bool sendSecondaryData = context.sendsSecondaryDataOnStatusExchange();
957 
958  // Create MPI Sends and Receives
959  std::vector<boost::mpi::request> requests;
960 
961  // STEP 5: Create the receives
962  std::vector<SyncStatus_Packet<Content>* > packetsRecd;

967  requests.push_back(world->irecv(source, AGENT_MOVED_AGENT, *packetToRecv));
968  packetsRecd.push_back(packetToRecv);
969  }
970

972  boost::ptr_list<SyncStatus_Packet<Content> >* packetsToSend = new boost::ptr_list<SyncStatus_Packet<Content> >;

975  // Agent Content
976  std::vector<Content>* content = new std::vector<Content>;
977  provider.provideContent(iter->second, *content);
978 
979  // Projection Info and Secondary Ids
980  std::map<std::string, std::vector<ProjectionInfoPacket*> >* projInfo = new std::map<std::string, std::vector<repast::ProjectionInfoPacket*> >;
981  std::set<AgentId>* secondaryIds = (sendSecondaryData ? new std::set<AgentId> : 0);
982  context.getProjectionInfo(iter->second, *projInfo, sendSecondaryData, secondaryIds);
983 
984  // Send the information for the secondary agents, too:
985  if(secondaryIds != 0){ // use 'sendSecondaryData' instead? Should be equivalent...
986  AgentRequest sidReq;
987  for(std::set<AgentId>::iterator sidIter = secondaryIds->begin(), sidIterEnd = secondaryIds->end(); sidIter != sidIterEnd; ++sidIter) sidReq.addRequest(*sidIter);
988  provider.provideContent(sidReq, *content); // Only their state data is needed, not any projection info
989  }
990

992  AgentExporterInfo* agentImporterInfoPtr = importer_exporter->getAgentExportInfo(iter->first);
993 
994  SyncStatus_Packet<Content>* packetToSend;
995  packetsToSend->push_back( packetToSend = new SyncStatus_Packet<Content>(content, projInfo, secondaryIds, agentImporterInfoPtr) );
996 
997  requests.push_back(world->isend(iter->first, AGENT_MOVED_AGENT, *packetToSend));
998  }
999  boost::mpi::wait_all(requests.begin(), requests.end());
1000  delete packetsToSend;
1001 
1002  importer_exporter->clearAgentExportInfo();
1003 
1004  // STEP 9: Remove the agents that are moving to other processes and are not needed here
1005  std::set<AgentId> agentsToKeep;
1006  context.getRequiredAgents(agentsToDrop, agentsToKeep, Projection<T>::SECONDARY);
1007 
1008  for(std::set<AgentId>::iterator idIter = agentsToDrop.begin(), idIterEnd = agentsToDrop.end(); idIter != idIterEnd; ++idIter) context.removeAgent(*idIter);
1009 
1010  // STEP 10: Insert the newly received agents that moved to this process and update exporters
1011  typename std::vector<SyncStatus_Packet<Content>* >::iterator packetIter;
1012  typename std::vector<SyncStatus_Packet<Content>* >::iterator packetIterEnd = packetsRecd.end();
1013  AgentRequest secondaryAgentsToRequest(rank_);
1014  for(packetIter = packetsRecd.begin(); packetIter != packetIterEnd; ++packetIter){
1015  std::vector<Content>* content = (*packetIter)->agentContentPtr;
1016  typename std::vector<Content>::iterator contentIter = content->begin();
1017  while(contentIter != content->end()){
1018  T* out = creator.createAgent(*contentIter);
1019  T* inContext = context.addAgent(out);
1020  if (inContext != out) { // Already contain the agent
1021  // If the agent is local on this rank, do nothing (the agent received must be a secondary agent)
1022  // If the agent is non-local on this rank
1023  if(inContext->getId().currentRank() != rank_){
1024  // If the arriving agent has a current rank equal to this rank, this
1025  // is an incoming, newly arrived local agent that already existed on this
1026  // process as a secondary agent; it should be updated and its currentRank in
1027  // its ID set to the local rank
1028  if(out->getId().currentRank() == rank_){
1029  updater.updateAgent(*contentIter);

1031  }

1033  // already exists as a non-local agent on this process; leave the original alone
1034  // and discard the new version.
1035  }
1036  delete out;
1037  }
1038  else{// Agent was not already on this rank and is not a new local agent; must process it as a new request
1039  if(out->getId().currentRank() != rank_) secondaryAgentsToRequest.addRequest(out->getId());
1040  }
1041  contentIter++;
1042  }
1043 
1044  // Update the importer/exporter to reflect the newly arrived local agents
1045  importer_exporter->incorporateAgentExporterInfo(*((*packetIter)->exporterInfoPtr));
1046  importer_exporter->clearExportToSpecificProc(rank_); // Export Info may include 'exports' to self; remove these
1047 
1048  }
1049

1051  if(sendSecondaryData) initiateAgentRequest(secondaryAgentsToRequest);
1052 
1053  // STEP 12: Set all the Projection Info, including graph edges, and clear the received data
1054  for(packetIter = packetsRecd.begin(); packetIter != packetIterEnd; ++packetIter){
1055  context.setProjectionInfo(*((*packetIter)->projectionInfoPtr));
1056  delete (*packetIter)->deleteExporterInfo(); // Exporter Info is only deleted from the received packets, not the sent ones...
1057  }
1058 
1059 }
