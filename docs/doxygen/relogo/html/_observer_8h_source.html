638 template<typename AgentType, typename FactoryFunctor>
639 AgentType* Observer::hatch(RelogoAgent* parent, FactoryFunctor agentCreator) {
640  const std::type_info* info = &(typeid(AgentType));
641  int agentTypeId = 0;
642  int id = 0;
643  TypeMapIterator iter = typeMap.find(info);
644  if (iter == typeMap.end()) {
645  agentTypeId = nextTypeId;
646  nextTypeId++;
647  } else {
648  agentTypeId = iter->second.first;
649  id = iter->second.second;
650  }
651 
652  AgentType* agent = agentCreator(repast::AgentId(id, _rank, agentTypeId), this);
653  agent->hatchCopy(parent);
654  context.addAgent(agent);
655  agent->setxy(parent->xCor(), parent->yCor());
656  typeMap[info] = std::make_pair(agentTypeId, id + 1);
657  return agent;
658 }

675 template<typename AgentType, typename FactoryFunctor>
676 int Observer::create(size_t count, FactoryFunctor agentCreator) {
677  const std::type_info* info = &(typeid(AgentType));
678  int agentTypeId = 0;
679  int id = 0;
680  TypeMapIterator iter = typeMap.find(info);
681  if (iter == typeMap.end()) {
682  agentTypeId = nextTypeId;
683  nextTypeId++;
684  } else {
685  agentTypeId = iter->second.first;
686  id = iter->second.second;
687  }
688 
689  DoubleUniformGenerator gen = Random::instance()->createUniDoubleGenerator(0, 360);
690  for (size_t i = 0; i < count; i++) {
691 // AgentType* agent = agentCreator(repast::AgentId(id + i, _rank, agentTypeId), this);
692  repast::AgentId agentid(id+i, _rank, agentTypeId);
693  agentid.currentRank(_rank);
694  AgentType* agent = agentCreator(agentid, this);
695  agent->heading((float) gen.next());
696  context.addAgent(agent);
697  }
698  typeMap[info] = std::make_pair(agentTypeId, id + count);
699  return agentTypeId;
700 }

821  AgentSet<AgentType>& outSet) {

827  for (size_t i = 0; i < inSet.size(); i++) {
828  RelogoAgent* agent = inSet[i];
829  if (spc->getDistanceSq(center, agent->location()) <= radiusSq && agent->getId().agentType() == typeId) {
830  outSet.add(static_cast<AgentType*> (const_cast<RelogoAgent*> (agent)));
831  }
832  }
